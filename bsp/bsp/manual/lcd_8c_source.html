364 static void st7579_command(unsigned char reg, unsigned char mask, unsigned char data, unsigned char h)
365 {
366 
367  /* rs ist standardmäßig auf 0 */
368 
369  if (h < 255) {
370  // Set H mode (instruction mode) if we're not already there.
371  if ( (st7579_funcset & 0x03) != h ) {
372  st7579_funcset = (st7579_funcset & 0xFC) | h;
373  st7579_command(ST7579_FUNCSET, ST7579_FUNCSET_MASK, st7579_funcset, 255);
374  }
375  }
376 
377  #if ( spiUSE_LOCKING )
378  {
379  spiLockChannel( lcdSPI_CHANNEL, lcdPID );
380  spiSelectSlave( lcdSPI_CHANNEL, spiSELECT_LCD, lcdPID );
381  spiTransmitByte( lcdSPI_CHANNEL, reg | ( data & mask ), lcdPID );
382  spiDeselectSlave( lcdSPI_CHANNEL, spiSELECT_LCD, lcdPID );
383  spiUnlockChannel( lcdSPI_CHANNEL, lcdPID );
384  }
385  #else
386  {
387  spiSelectSlave( lcdSPI_CHANNEL, spiSELECT_LCD );
388  spiTransmitByte( lcdSPI_CHANNEL, reg | ( data & mask ) );
389  spiDeselectSlave( lcdSPI_CHANNEL, spiSELECT_LCD );
390  }
391  #endif
392 
393 }

623  * @brief Initialisierung des Moduls lcd.
624  *
625  * @details Führt alle nötigen Initialisierungen aus.
626  *
627  * @return Wenn das Initialiseren erfolgreich abgeschlossen
628  * wurde, gibt die Fkt. #lcdERROR_NONE zurück. Wenn
629  * während der Initialisierung ein Fehler auftritt,
630  * wird eine negative Zahl zurückgegeben (siehe
631  * #lcdERROR_SPI_INIT_FAILED).
632  */
633 int lcdInitialize( void )
634 {
635 
636  if ( isInitialized )
637  {
638  return lcdERROR_NONE;
639  }
640 
641  if ( spiInitializeChannel( lcdSPI_CHANNEL ) != spiERROR_NONE )
642  {
643  return lcdERROR_SPI_INIT_FAILED;
644  }
645 
646  /* BL_ENA (Backlight-Enable)
647  als GPI0-Ausgangspin
648  konfigurieren und mit
649  1 initialisieren.
650  0 -> Hintergrundbeleuchtung aus,
651  1 -> Hintergrundbeleuchtung ein. */
652  PORT2.PODR.BIT.B1 = 1;
653  PORT2.PMR.BIT.B1 = 0;
654  PORT2.PDR.BIT.B1 = 1;
