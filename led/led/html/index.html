Dokumentation der Headerdatei led.h
Installation
Um die in der Headerdatei led.h definierten Makros zu Kontrolle der zwölf LEDs LED4 bis LED15 des YRDKRX63N nutzen zu können, müssen Sie die Headerdatei led.h in ihr Projektverzeichnis kopieren und inkludieren.

Beispielprogramm
#include <limits.h>
#include "led.h"
static void delay(void);
void main(void) {
int step;
ledInitialize();
for ( step = 0; ; step = (++step) % 12 ) {
switch ( step ) {
case 0 :
ledTurnOn(LED4);
ledTurnOff(LED15);
delay();
break;
case 1 :
ledTurnOn(LED5);
ledTurnOff(LED4);
delay();
break;
case 2 :
ledTurnOn(LED6);
ledTurnOff(LED5);
delay();
break;
case 3 :
ledTurnOn(LED7);
ledTurnOff(LED6);
delay();
break;
case 4 :
ledTurnOn(LED8);
ledTurnOff(LED7);
delay();
break;
case 5 :
ledTurnOn(LED9);
ledTurnOff(LED8);
delay();
break;
case 6 :
ledTurnOn(LED10);
ledTurnOff(LED9);
delay();
break;
case 7 :
ledTurnOn(LED11);
ledTurnOff(LED10);
delay();
break;
case 8 :
ledTurnOn(LED12);
ledTurnOff(LED11);
delay();
break;
case 9 :
ledTurnOn(LED13);
ledTurnOff(LED12);
delay();
break;
case 10:
ledTurnOn(LED14);
ledTurnOff(LED13);
delay();
break;
case 11:
ledTurnOn(LED15);
ledTurnOff(LED14);
delay();
break;
}
}
}
static void delay(void) {
int i;
for ( i = 0; i < INT_MAX / 10000000; i++ ) {
asm("nop");
}
}
