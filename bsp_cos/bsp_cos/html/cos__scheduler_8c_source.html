620 int8_t COS_RunScheduler(void)
621 {
622 
623 
624  Node_t *pt=NULL;
625  uint16_t t_Ticks;
626 
627  //DebugCode(_msg("RunScheduler,prio based\r\n"););
628 
629  pt = root_g; /* first task, highest prio */
630  while(1) /* loop forever */
631  { /* time to run? */
632  t_Ticks = _gettime_Ticks();
633  /* time wrap around is ok, time difference will be right... */
634  if(((uint16_t)(t_Ticks - pt->task_pt->lastActivationTime_Ticks) >=
635  pt->task_pt->sleepTime_Ticks)&&
636  (pt->task_pt->state == TASK_STATE_READY))
637  { pt->task_pt->lastActivationTime_Ticks = t_Ticks;
638  /* when the task function runs to its very end, the task will be deleted:
639  it will be removed from the list, and the task struct will be freed,
640  i.e. pt->task_pt is no longer valid.
641  */
642  pt->task_pt->sleepTime_Ticks = 0; // Bugfix 22.10.2015: must be specified by task!
643  pt->task_pt->func(pt->task_pt); /* call task function, must not block! */
644  pt = root_g; /* next: check task with highest prio */
645  }
646  else
647  { pt = pt->next_pt; /* check next task in list */
648  if(NULL==pt)
649  { pt = root_g; /* treat linear list as ring list */
650  }
651  }
652  }return 0;
653 }
