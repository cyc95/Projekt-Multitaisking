125 uint8_t COS_FifoCreate(CosFifo_t *q, uint8_t slotSize, uint8_t nSlots)
126 {
127  /* create buffer */
128  q->buffer = (char *) malloc(slotSize * nSlots * sizeof(char));
129  if(NULL == q->buffer)
130  { DebugCode(_msg("FifoCreate:malloc!"););
131  return -1;
132  }
133  q->maxSlots = nSlots;
134  q->slotSize = slotSize;
135  q->rIndex = 0; /* empty queue */
136  q->wIndex = 0;
137  q->usedSlots = 0;
138  if(0!= COS_SemCreate(&(q->rSema), 0)) // nothing to read yet
139  { DebugCode(_msg("FifoCreate:SemCreate!"););
140  return -1;
141  }
142  if(0!= COS_SemCreate(&(q->wSema), nSlots)) // all slots are still free
143  { DebugCode(_msg("FifoCreate:SemCreate!"););
144  return -1;
145  }
146  q->isInitialized = 1;
147  return 0;
148 }

311 int8_t _qWriteSingleSlot(CosFifo_t *q, const char *data)
312 { int8_t retval = 0;
313  if(q->isInitialized == 0)
314  { DebugCode(_msg("_qWriteSingleSlot:not init"););
315  return -1;
316  }
317  /* if FIFO is full, return 0 without writing any data */
318  if(q->usedSlots >= q->maxSlots)
319  { retval = 0; /* FIFO is full, don't write */
320  }
321  else /* at least one slot is free, write data */
322  { retval = 1;
323  memcpy(&(q->buffer[q->wIndex]), data, q->slotSize); /* copy to FIFO */
324  q->wIndex += q->slotSize; /* next slot */
325  q->wIndex %= (q->maxSlots * q->slotSize); /* circular buffer */
326  q->usedSlots += 1;
327  COS_SEM_SIGNAL(&(q->rSema)); // unblock tasks that wait for reading,
328  }
329  return retval;
330 }

cos_data_fifo.h
Daten-FIFO fuer COS auf Atmel. Der FIFO benutzt dynamische Speicherverwaltung (malloc()). Es kann nur ein Sorte Daten gespeichert werden. Die maximale Anzahl von Speicherplaetzen (Slots) ist 255 wegen der Verwendung von 1 Byte Indexvariablen. Ein Slot kann maximal 256 Byte Speichern.
