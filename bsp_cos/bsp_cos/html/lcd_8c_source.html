365 {
366

373  st7579_command(ST7579_FUNCSET, ST7579_FUNCSET_MASK, st7579_funcset, 255);

784 
785  /* 0 ... 0xFF drei mal durchlaufen. */
786  for ( i = 0; i <= 0x2FF; i++ )
787  {
788  if ( i % lcdWIDTH == 0 && i )
789  {
790  /* Der ST7579 verfügt über
791  102 Spalten, der LCD aber
792  nur über 96 (lcdWIDTH),
793  also 6 Dummy-Writes um in
794  die nächste Zeile zu gelangen
795  bevor die nächsten Daten
796  gesendet werden. */
797  for ( j = 0; j < 6; j++ )
798  {
799  st7579_write( ( unsigned char * ) &j, 1 );
800  }
801  }
802  st7579_write( ( unsigned char * ) &i, 1 );
803  }
804 
805  /* Der ST7579 hat 68 Zeilen, der LCD
806  aber nur 64, also 6 Dummy-Writes um in
807  die nächste Zeile zu gelangen und 102
808  Dummy-Writes um an das Ende der Zeile zu
809  gelangen. */
810  for ( i = 0; i < 6 + 102; i++ )
811  {
812  st7579_write( ( unsigned char * ) &i, 1 );
813  }
814 
815  /* Was jetzt gesendet werden würde, würde wieder
816  oben links erscheinen. */
817 
818  return lcdERROR_NONE;
819 
820  }
