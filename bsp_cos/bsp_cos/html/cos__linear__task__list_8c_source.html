176 { Node_t *pt=NULL;
177  Node_t *predecessor_pt=NULL;
178 
179  pt = _searchTaskInList(root_pt, task_pt);
180  if(NULL == pt)
181  { DebugCode(_msg("_unlink...task not found"););
182  return root_pt; /* don't change the list! */
183  }
184  /* is it the first node in the list? root_pt has to be changed... */
185  if(pt == root_pt)
186  { pt = pt->next_pt; /* second node in list */
187  free(root_pt); /* free the first node, don't touch the task! */
188  return pt; /* the old second element now is the first */
189  }
190  /* node exists, is not the first list element, should have a predecessor... */
191  predecessor_pt = _searchPredecessorTaskInList(root_pt, task_pt);
192  if(NULL == predecessor_pt)
193  { DebugCode(_msg("_unlink...pred.task not found"););
194  return root_pt; /* don't change the list! */
195  }
196  /* task node and its predecessor have been found. Unlink task node: */
197  predecessor_pt->next_pt = pt->next_pt;
198  free(pt); /* free the node, don't touch the task! */
199  return root_pt; /* old first list element still is the first */

428 void _sortLinearListPrio(Node_t *root_pt)
429 {
430 
431  int noOfSwaps;
432  CosTask_t *temp_pt;
433  Node_t *pt;
434 
435  if (root_pt==NULL) return; /* empty list */
436 
437  /* bubblesort */
438  do
439  { noOfSwaps=0;
440  pt = root_pt;
441  while(pt->next_pt != NULL) // run through linear list
442  { if((pt->task_pt)->prio < ((pt->next_pt)->task_pt)->prio)
443  { /* swap the 'payloads' of the nodes*/
444  noOfSwaps++;
445  temp_pt = pt->task_pt; /* swap payload of nodes */
446  pt->task_pt = (pt->next_pt)->task_pt;
447  (pt->next_pt)->task_pt = temp_pt;
448  }
449  pt = pt->next_pt;
450  }
451  } while(noOfSwaps >0);
